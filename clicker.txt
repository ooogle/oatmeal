Shinies
	*luckyOat
		on click:log Woop
		movement:onRight moveLeft fade bounce:0.05
		frequency:60
		frequency variation:30
		icon:https://amindfullmom.com/wp-content/uploads/2018/02/Instant-Pot-Creamy-Oatmeal_.jpg
		class:bigButton
		on click:
			$boost=1
			if (have clover) $boost=4
			if (chance(25%))
				//get at least 7, or between 1 and 3 minutes of our bunny production
				$amount=max(7,random(oats:ps*60*1,oats:ps*60*3))*$boost
				toast All dem oat.
				yield $amount oats
			else
				$amount=1*$boost
				toast all dem oatz
				yield $amount goldenOat
			end
		end
